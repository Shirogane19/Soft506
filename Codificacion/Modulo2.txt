    Function obtenerMayorGenero(generoDioses As Boolean()) As Integer
        Dim contadorHombres As Integer
        Dim contadorMujeres As Integer
        Dim mensaje As String

        For i As Integer = 0 To 8
            If (generoDioses(i) = False) Then
                contadorHombres = contadorHombres + 1
            Else
                contadorMujeres = contadorMujeres + 1
            End If
        Next

    Function obtenerDatosDiosesPorGenero(nombreDioses As String(), generoDioses As Boolean(), numeroHijos As Integer())

        Dim contadorMujeres As Integer
        For i As Integer = 0 To 8

            If (generoDioses(i) = False) Then
                genero = "masculino"
                datosDiosesHombres(contadorHombres) = nombreDioses(i) + " " + genero + " " + numeroHijos(i).ToString()

            Else
                genero = "femenino"
                datosDiosesMujeres(contadorMujeres) = nombreDioses(i) + " " + genero + " " + numeroHijos(i).ToString()
                contadorMujeres = contadorMujeres + 1
            End If
        Next

        For i As Integer = 0 To contadorHombres - 1
            If (String.IsNullOrEmpty(datosDiosesHombres(i)) = False) Then
                Console.WriteLine("/n")
                Console.WriteLine(datosDiosesHombres(i))
                Console.Write("*------------------------------------*")
            End If
        Next

        For i As Integer = 0 To contadorMujeres - 1

                Console.WriteLine("/n")
                Console.WriteLine(datosDiosesMujeres(i))
